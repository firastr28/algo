ALGORITHM check-algo
VAR
    i,j,k:INTEGER:=0;
    x,y,z:INTEGER:=0;
    ch:CHAR;
BEGIN
 WHILE (i=0;i=<length.ch) DO
        Read(ch)
        
                IF (i=length.ch) THEN
                  x=x+1;
                 END_IF
                IF (ch[i]=" ") THEN
                        y=y+1;
                END_IF 
                IF (if ch[i]="A"or "E" or "I" or "O" or "U" or "Y") THEN
                        z=z+1;                    
                END_IF   
                    END_WHILE
END
